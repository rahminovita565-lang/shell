/* ----------------- HELPERS ----------------- */
function hfs($bytes){
    $u=["B","KB","MB","GB","TB"];
    $i=0;
    while($bytes>=1024 && $i < count($u)-1){ $bytes/=1024; $i++; }
    return round($bytes,2).' '.$u[$i];
}
function esc($s){ return htmlspecialchars($s, ENT_QUOTES); }

// TERMINAL COMMAND HANDLER
// =============================================================================

function run_terminal_safe($cmd, $workdir) {
    $cmd = trim($cmd);
    if (empty($cmd)) return '';
    
    // Split command
    $parts = preg_split('/\s+/', $cmd);
    $base = strtolower($parts[0]);
    
    $out = '';
    
    switch ($base) {
        case 'ls':
        case 'dir':
            $files = scandir($workdir);
            foreach ($files as $f) {
                if ($f === '.' || $f === '..') continue;
                $p = $workdir . DIRECTORY_SEPARATOR . $f;
                $type = is_dir($p) ? '[DIR]' : '[FILE]';
                $size = is_file($p) ? ' (' . filesize($p) . ' bytes)' : '';
                $out .= "$type $f$size\n";
            }
            break;
            
        case 'pwd':
            $out = $workdir;
            break;
            
        case 'cat':
            if (!isset($parts[1])) {
                $out = "Usage: cat filename";
                break;
            }
            $file = $workdir . DIRECTORY_SEPARATOR . basename($parts[1]);
            if (is_file($file)) {
                $out = file_get_contents($file);
            } else {
                $out = "File not found: " . basename($parts[1]);
            }
            break;
            
        case 'echo':
            $out = implode(' ', array_slice($parts, 1));
            break;
            
        case 'whoami':
            $out = get_current_user();
            break;
            
        case 'php':
            if (isset($parts[1]) && $parts[1] === '-v') {
                $out = phpversion();
            } else {
                $out = "PHP " . phpversion();
            }
            break;
            
        case 'help':
            $out = "Available commands:\n";
            $out .= "ls/dir  - List files\n";
            $out .= "pwd     - Print working directory\n";
            $out .= "cat     - Display file content\n";
            $out .= "echo    - Echo text\n";
            $out .= "whoami  - Current user\n";
            $out .= "php -v  - PHP version\n";
            $out .= "help    - Show this help\n";
            break;
            
        default:
            $out = "Command not recognized. Type 'help' for available commands.";
    }
    
    return $out;
}
